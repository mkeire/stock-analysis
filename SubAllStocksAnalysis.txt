Sub AllStocksAnalysis()
    Worksheets("All Stocks Analysis").Activate

    Range("A1").Value = "All Stocks (2018)"

    'Create header row
    Cells(3, 1).Value = "Ticker"
    Cells(3, 2).Value = "Total Daily Volume"
    Cells(3, 3).Value = "Return"
    
    'Create an array for tickers
    Dim tickers(12) As String
    tickers(0) = "AY"
    tickers(1) = "CSIQ"
    tickers(2) = "DQ"
    tickers(3) = "ENPH"
    tickers(4) = "FSLR"
    tickers(5) = "HASI"
    tickers(6) = "JKS"
    tickers(7) = "RUN"
    tickers(8) = "SEDG"
    tickers(9) = "SPWR"
    tickers(10) = "TERP"
    tickers(11) = "VSLR"
    
    Dim startingPrice As Single
    Dim endingPrice As Single
    
    Worksheets("2018").Activate
    
    'Determine number of rows to loop through
    RowCount = Cells(Rows.Count, "A").End(xUp).Row
    
    For i = 0 To 11
    
        ticker = tickers(i)
        totalVolume = 0
        
        Worksheets("2018").Activate
        For j = 2 To RowCount
        
            'Determine stock's total volume traded
            If Cells(j, 1).Value = ticker Then
                totalVolume = totalVolume + Cells(j, 8).Value
            End If
            
            'Get starting price
            If Cells(j - 1, 1).Value <> ticker And Cells(j, 1).Value = ticker Then
                startingPrice = Cells(j, 6).Value
            End If
            
            'Get ending price
            If Cells(j, 1).Value <> ticker And Cells(j, 1).Value = ticker Then
                endingPrice = Cells(j, 6).Value
            End If
            
        Next j
            
    Next i
    
    'Formatting
    Worksheets("All Stock Analysis").Activate
    Range("A3:C3").Font.Bold = True
    Range("A3:C3").Border(xlEdgeBottom).LineStyle = xlContinuous
    Range("B4:B15").NumberFormat = "$#,##0"
    Range("C4:C15").NumberFormat = "0.00%"
    Columns("B").AutoFit
    
    dataRowStart = 4
    dataRowEnd = 15
    For i = dataRowStart To dataRowEnd

        If Cells(4, 3) > 0 Then

            'Color the cell green
            Cells(4, 3).Interior.Color = vbGreen

        ElseIf Cells(4, 3) < 0 Then

            'Color the cell red
            Cells(4, 3).Interior.Color = vbRed

        Else

            'Clear the cell color
            Cells(4, 3).Interior.Color = xlNone

        End If

    Next i
    
End Sub
